def sortir(objet, meuble):
    # pré-requis : au moins une main est libre
    ouvrir(meuble) 
    prendre(objet)
    fermer(meuble)
    poser(objet)
    
def ranger(objet, meuble):
    # pré-requis : l'objet est sorti et les deux mains sont libres
    prendre(objet)
    ouvrir(meuble) 
    poser(objet) 
    fermer(meuble)
